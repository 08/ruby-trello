def update_fields(fields)
  attributes[:yellow]  = fields['yellow']
  attributes[:red] = fields['red']
  attributes[:orange] = fields['orange']
  attributes[:green] = fields['green']
  attributes[:purple] = fields['purple']
  attributes[:blue] = fields['blue']
  self
end
